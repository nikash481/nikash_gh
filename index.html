C Language Codes & Questions

1. Macros

Used to define constants or code fragments:

#define PI 3.14159
#define SQUARE(x) ((x) * (x))

2. File Inclusion

Used to include standard or user-defined files:

#include <stdio.h>
#include "myheader.h"

3. Conditional Compilation

Used to compile certain parts of code conditionally:

#define DEBUG

#ifdef DEBUG
    printf("Debug mode is ON\n");
#endif

4. Other Directives

Various utility directives:

#undef PI
#error "Custom error message"
#pragma once
Examples:

1. Macros

#include <stdio.h>
#define PI 3.14
#define SQUARE(x) x*x

int main() {
    #ifdef PI
        printf("PI is defined as: %.2f\\n", PI);
    #endif
    printf("Square of 5 is: %d\\n", SQUARE(5));
    return 0;
}

2. Tokens

#include <stdio.h>

int main() {
    int num = 10;
    float pi = 3.14;
    printf("Value is: %d\\n", num);
    return 0;
}

3. Input/Output Functions

#include <stdio.h>

int main() {
    char ch;
    char name[20];

    printf("Enter a character: ");
    ch = getchar();
    putchar(ch);

    printf("\\nEnter your name: ");
    gets(name);
    puts(name);

    int a;
    printf("\\nEnter a number: ");
    scanf("%d", &a);
    printf("Number is: %d\\n", a);

    return 0;
}

4. switch-case and goto

#include <stdio.h>

int main() {
    int choice;
    printf("Enter a number (1-3): ");
    scanf("%d", &choice);

    switch (choice) {
        case 1: printf("You chose One\\n"); break;
        case 2: printf("You chose Two\\n"); break;
        case 3: goto label;
        default: printf("Invalid choice\\n");
    }

    label:
        printf("Jumped using goto!\\n");

    return 0;
